YOUTUBE

axios
	.get(
		`https://youtube.googleapis.com/youtube/v3/search?part=snippet&channelId=${elvnId}&maxResults=46&order=date&type=video&key=${process.env.YOUTUBE_API_KEY}`
	)
	.then(({ data }) => {
		let videoIds = [];

		data.items.forEach((element) => {
			if (element.id.kind === 'youtube#video') {
				videoIds.push(element.id.videoId);
			}
		});

		console.log(videoIds, videoIds.length);
	})
	.catch((err) => {
		console.log(err);
	});
